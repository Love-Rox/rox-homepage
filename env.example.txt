# Environment Variables for Contact Form

# Cloudflare Turnstile Keys
# Get these from: https://dash.cloudflare.com/?to=/:account/turnstile
VITE_TURNSTILE_SITE_KEY=your_turnstile_site_key_here
TURNSTILE_SECRET_KEY=your_turnstile_secret_key_here

# Discord webhook for contact form notifications
# Recommended: target a Forum channel — each submission becomes its own thread,
# with a copy-paste reply template posted as a follow-up message in the same thread.
# A regular text channel also works, but the reply template will appear as the next
# top-level message rather than inside a thread.
DISCORD_CONTACT_WEBHOOK_URL=https://discord.com/api/webhooks/your_id/your_token

# For testing, you can use Cloudflare's test keys:
# VITE_TURNSTILE_SITE_KEY=1x00000000000000000000AA
# TURNSTILE_SECRET_KEY=1x0000000000000000000000000000000AA
